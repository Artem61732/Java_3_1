git --version 				- актуальная версия
git --help				- список всех комманд
dir 					- список текущик папок в данной директории
cd 					- перейти в директорию
git init				- инициализация репозитория
cd ..					- вернуться к папке назад
git status				- файлы которые сейчас отслеживаются
git add назв_файла			- добавить файл в репозиторий
git add . 				- добавить все измененные файлы
git rm --cached назв_файла		- перестать следить за файлом
git commit -m "назв_коммита"		- добавление коммита (обязателен для создания ветки)
.gitignore				- в него нужно написать названия файлов, которые нужно заигнорить,
				  	  но сам gitignore нужно затрекать

git branch				- просмотр веток 
git branch назв_ветки			- создание новой ветки
git branch -D назв_ветки		- удалить ветку
git checkout назв_ветки 		- переход к ветке
git checkout -b назв_ветки		- одновременно создать ветку и сразу перейти на неё
git merge назв_ветки 			- объединение текущей ветки с указанной

git config --global user.name "имя"	- установить имя на гитхаб
git config --global user.name		- отобразить имя на гитхаб
git config --global user.email "мэил"	- установить мэил на гитхаб
git config --global user.email		- отобразить мэил на гитхаб
git branch -M main 			- переименовать ветку master в main

git remote add origin https://github.com/Artem61732/Java.git	- привязка локального репозитория к удаленному
git push -u origin master		- залить на сервер (главную ветку)
git push				- залить на сервер (обновленный файл)
git remote -v				- вы получите список дескрипторов и связанных URL-адресов
git remote remove origin		- удалить старые ссылки на репозиторий гитхаба
